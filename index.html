        // JavaScript para o toggle do menu de navegação em telas menores

        document.getElementById('nav-toggle').onclick = function () {

        // JavaScript para as animações suaves ao rolar a página

        const sections = document.querySelectorAll('section[id]'); // Seleciona todas as seções com um ID

        const animateOnScroll = () => {

            sections.forEach(section => {

                const sectionTop = section.getBoundingClientRect().top; // Posição do topo da seção em relação à viewport

                const windowHeight = window.innerHeight; // Altura da janela de visualização

                // Se o topo da seção estiver dentro de 75% da altura da janela, adiciona as classes de animação

                if (sectionTop < windowHeight * 0.75) {

                    section.classList.add('animate__animated', 'animate__fadeInUp');

                    // Pega o atraso da animação definido no atributo data-animate-delay (se existir)

                    const delay = section.dataset.animateDelay || '0s';

                    section.style.animationDelay = delay;

                }

            });

        };


        // Adiciona listeners para o evento de scroll e load para disparar as animações

        window.addEventListener('scroll', animateOnScroll);

        window.addEventListener('load', animateOnScroll); // Para animar elementos que já estão visíveis ao carregar a página

    </script>
